47      /**
48       * Reads in the specified POM and converts it to a Model.
49       *
50       * @param file the pom.xml file
51       * @return returns a
52       * @throws AnalysisException is thrown if there is an exception extracting or parsing the POM
53       * {@link org.owasp.dependencycheck.jaxb.pom.generated.Model} object
54       */
55      public static Model readPom(File file) throws AnalysisException {
56          Model model = null;
57          try {
58              PomParser parser = new PomParser();
59              model = parser.parse(file);
60          } catch (PomParseException ex) {
61              final String msg = String.format("Unable to parse pom '%s'", file.getPath());
62              LOGGER.log(Level.WARNING, msg);
63              LOGGER.log(Level.FINE, "", ex);
64              throw new AnalysisException(ex);
65          } catch (IOException ex) {
66              final String msg = String.format("Unable to parse pom '%s'(IO Exception)", file.getPath());
67              LOGGER.log(Level.WARNING, msg);
68              LOGGER.log(Level.FINE, "", ex);
69              throw new AnalysisException(ex);
70          } catch (Throwable ex) {
71              final String msg = String.format("Unexpected error during parsing of the pom '%s'", file.getPath());
72              LOGGER.log(Level.WARNING, msg);
73              LOGGER.log(Level.FINE, "", ex);
74              throw new AnalysisException(ex);
75          }
76          return model;
77      }
78  
79      /**
80       * Retrieves the specified POM from a jar file and converts it to a Model.
81       *
82       * @param path the path to the pom.xml file within the jar file
83       * @param jar the jar file to extract the pom from
84       * @return returns a
85       * @throws AnalysisException is thrown if there is an exception extracting or parsing the POM
86       * {@link org.owasp.dependencycheck.jaxb.pom.generated.Model} object
87       */
88      public static Model readPom(String path, JarFile jar) throws AnalysisException {
89          final ZipEntry entry = jar.getEntry(path);
90          Model model = null;
91          if (entry != null) { //should never be null
92              try {
93  //                final NonClosingStream stream = new NonClosingStream(jar.getInputStream(entry));
94  //                final InputStreamReader reader = new InputStreamReader(stream, "UTF-8");
95  //                final InputSource xml = new InputSource(reader);
96  //                final SAXSource source = new SAXSource(xml);
97                  final PomParser parser = new PomParser();
98                  model = parser.parse(jar.getInputStream(entry));
99                  LOGGER.fine(String.format("Read POM %s", path));
100             } catch (SecurityException ex) {
101                 final String msg = String.format("Unable to parse pom '%s' in jar '%s'; invalid signature", path, jar.getName());
102                 LOGGER.log(Level.WARNING, msg);
103                 LOGGER.log(Level.FINE, null, ex);
104                 throw new AnalysisException(ex);
105             } catch (IOException ex) {
106                 final String msg = String.format("Unable to parse pom '%s' in jar '%s' (IO Exception)", path, jar.getName());
107                 LOGGER.log(Level.WARNING, msg);
108                 LOGGER.log(Level.FINE, "", ex);
109                 throw new AnalysisException(ex);
110             } catch (Throwable ex) {
111                 final String msg = String.format("Unexpected error during parsing of the pom '%s' in jar '%s'", path, jar.getName());
112                 LOGGER.log(Level.WARNING, msg);
113                 LOGGER.log(Level.FINE, "", ex);
114                 throw new AnalysisException(ex);
115             }
116         }
117         return model;
118     }

120     /**
121      * Reads in the pom file and adds elements as evidence to the given dependency.
122      *
123      * @param dependency the dependency being analyzed
124      * @param pomFile the pom file to read
125      * @throws AnalysisException is thrown if there is an exception parsing the pom
126      */
127     public static void analyzePOM(Dependency dependency, File pomFile) throws AnalysisException {
128         final Model pom = PomUtils.readPom(pomFile);
129 
130         String groupid = pom.getGroupId();
131         String parentGroupId = null;
132 
133         if (pom.getParentGroupId() != null) {
134             parentGroupId = pom.getParentGroupId();
135             if ((groupid == null || groupid.isEmpty()) && parentGroupId != null && !parentGroupId.isEmpty()) {
136                 groupid = parentGroupId;
137             }
138         }
139         if (groupid != null && !groupid.isEmpty()) {
140             dependency.getVendorEvidence().addEvidence("pom", "groupid", groupid, Confidence.HIGHEST);
141             dependency.getProductEvidence().addEvidence("pom", "groupid", groupid, Confidence.LOW);
142             if (parentGroupId != null && !parentGroupId.isEmpty() && !parentGroupId.equals(groupid)) {
143                 dependency.getVendorEvidence().addEvidence("pom", "parent-groupid", parentGroupId, Confidence.MEDIUM);
144                 dependency.getProductEvidence().addEvidence("pom", "parent-groupid", parentGroupId, Confidence.LOW);
145             }
146         }
147         String artifactid = pom.getArtifactId();
148         String parentArtifactId = null;
149         if (pom.getParentArtifactId() != null) {
150             parentArtifactId = pom.getParentArtifactId();
151             if ((artifactid == null || artifactid.isEmpty()) && parentArtifactId != null && !parentArtifactId.isEmpty()) {
152                 artifactid = parentArtifactId;
153             }
154         }
155         if (artifactid != null && !artifactid.isEmpty()) {
156             if (artifactid.startsWith("org.") || artifactid.startsWith("com.")) {
157                 artifactid = artifactid.substring(4);
158             }
159             dependency.getProductEvidence().addEvidence("pom", "artifactid", artifactid, Confidence.HIGHEST);
160             dependency.getVendorEvidence().addEvidence("pom", "artifactid", artifactid, Confidence.LOW);
161             if (parentArtifactId != null && !parentArtifactId.isEmpty() && !parentArtifactId.equals(artifactid)) {
162                 dependency.getProductEvidence().addEvidence("pom", "parent-artifactid", parentArtifactId, Confidence.MEDIUM);
163                 dependency.getVendorEvidence().addEvidence("pom", "parent-artifactid", parentArtifactId, Confidence.LOW);
164             }
165         }
166         //version
167         String version = pom.getVersion();
168         String parentVersion = null;
169         if (pom.getParentVersion() != null) {
170             parentVersion = pom.getParentVersion();
171             if ((version == null || version.isEmpty()) && parentVersion != null && !parentVersion.isEmpty()) {
172                 version = parentVersion;
173             }
174         }
175         if (version != null && !version.isEmpty()) {
176             dependency.getVersionEvidence().addEvidence("pom", "version", version, Confidence.HIGHEST);
177             if (parentVersion != null && !parentVersion.isEmpty() && !parentVersion.equals(version)) {
178                 dependency.getVersionEvidence().addEvidence("pom", "parent-version", version, Confidence.LOW);
179             }
180         }
181 
182         final String orgName = pom.getOrganization();
183         if (orgName != null && !orgName.isEmpty()) {
184             dependency.getVendorEvidence().addEvidence("pom", "organization name", orgName, Confidence.HIGH);
185         }
186         final String pomName = pom.getName();
187         if (pomName != null && !pomName.isEmpty()) {
188             dependency.getProductEvidence().addEvidence("pom", "name", pomName, Confidence.HIGH);
189             dependency.getVendorEvidence().addEvidence("pom", "name", pomName, Confidence.HIGH);
190         }
191 
192         if (pom.getDescription() != null) {
193             final String description = pom.getDescription();
194             if (description != null && !description.isEmpty()) {
195                 JarAnalyzer.addDescription(dependency, description, "pom", "description");
196             }
197         }
198         JarAnalyzer.extractLicense(pom, null, dependency);
199     }
